sql {
    driver = "rlm_sql_postgresql"
    dialect = "postgresql"

    # Connection info ??? uses environment variables set in docker-compose.yml
    server = "$ENV{ENV_RADIUS_DB_HOST}"
    port = 5432
    login = "$ENV{ENV_RADIUS_DB_USER}"
    password = "$ENV{ENV_RADIUS_DB_PASS}"
    radius_db = "$ENV{ENV_RADIUS_DB_NAME}"

    # Standard FreeRADIUS table names (matching Prisma schema)
    acct_table1 = "radacct"
    acct_table2 = "radacct"
    postauth_table = "radpostauth"
    authcheck_table = "radcheck"
    groupcheck_table = "radgroupcheck"
    authreply_table = "radreply"
    groupreply_table = "radgroupreply"
    usergroup_table = "radusergroup"

    # Delete stale sessions on startup
    delete_stale_sessions = yes

    # Connection pool
    pool {
        start = 5
        min = 4
        max = 10
        spare = 3
        uses = 0
        retry_delay = 30
        lifetime = 0
        idle_timeout = 60
    }

    # Read NAS clients from database (nas table)
    read_clients = yes
    client_table = "nas"

    # Group membership attribute
    group_attribute = "SQL-Group"

    # Load standard PostgreSQL queries
    $INCLUDE ${modconfdir}/${.:name}/main/${dialect}/queries.conf
}
